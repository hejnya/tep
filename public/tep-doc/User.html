    Returns kominace sloupců, něco jako “#{name} #{last_name}” pokud dané spouce existují

    # File app/models/user.rb, line 81
def full_name
  if !name.nil?  && !last_name.nil?
    "#{name} #{last_name}"
  elsif !name.nil? 
    "#{name}"
  elsif !last_name.nil? 
    "#{last_name}"
  else
    email.split('@',2)[0]
  end
end
