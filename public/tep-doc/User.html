def full_name
  if !name.nil?  && !last_name.nil?
    "#{name} #{last_name}"
  elsif !name.nil? 
    "#{name}"
  elsif !last_name.nil? 
    "#{last_name}"
  else
    email.split('@',2)[0]
  end
end
